```typescript
 interface Counter {
   count: number;
   increment(): this;
   reset(): this;
 }

 class MyCounter implements Counter {
   count = 0;
  
   increment() {
     this.count++;
     return this;
   }
  
   reset() {
     this.count = 0;
     return this;
   }
 }

 const counter = new MyCounter();

 counter.increment().increment().reset();
```
